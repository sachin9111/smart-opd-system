pipeline {

    agent any

    tools {
        maven 'Maven3'
        jdk 'JDK17'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/sachin9111/smart-opd-system.git'
            }
        }

        stage('Build Discovery Server') {
            steps {
                dir('backend/discovery-server') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build API Gateway') {
            steps {
                dir('backend/api-gateway') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Auth Service') {
            steps {
                dir('backend/auth-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build User Service') {
            steps {
                dir('backend/user-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Doctor Service') {
            steps {
                dir('backend/doctor-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Appointment Service') {
            steps {
                dir('backend/appointment-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Queue Service') {
            steps {
                dir('backend/queue-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Patient Service') {
            steps {
                dir('backend/patient-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Notification Service') {
            steps {
                dir('backend/notification-service') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }
		
		stage('Install Angular Dependencies') {

			steps {

				dir('frontend/smart-opd-ui') {

					sh 'npm install'

				}

			}

		}

		stage('Build Angular') {

			steps {

				dir('frontend/smart-opd-ui') {

					sh 'npm run build'

				}

			}

		}

    }
}
